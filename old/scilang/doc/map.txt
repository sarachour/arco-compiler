process MRNA {
   Inputs:
      PreMRNA from PreMRNA : PreMRNA;
      SplicingActivator activates: Protein,Metabolite;
      RecombinaseInhibitor inhibits: Protein,Metabolite;
      Dicer inhibits: MRNA;
      Drosha inhibits: MRNA;
      Apatazyme inhibits: MRNA;

      SmallActivator excites: Metabolite;
      SmallInhibitor inhibits: Metabolite;
      3UTR excites: MRNA;
      RNAi excites : MRNA;
      RISC excites : MRNA;
      RNAase excites: MRNA;
   Outputs:
      MRNA: MRNA;
   Parameters:
   Structure:
      AM1 = AnalogMultiplier({});
      AM2 = AnalogMultiplier({});
      DC = DependentCurrentGenerator({});
      CAP = Capacitor({});
      VR = VariableResistor({});

      SplicingActivator -> AM1;
      RecombinaseInhibitor -> AM2;
      AM1,AM2 -> DC;

      RESM1 = AnalogMultiplier();
      RESM2 = AnalogMultiplier();
      1/Dicer*1/Crosha*1/Apatazyme*1/SmallInhibitor -> RSM1;
      3UTR*SmallActivator*RNAi*RISC -> RSM2;
      RSM1,RSM2,RNAase -> VR;


      STORE=DC->CAP;
      DEGRADE=STORE->VR;
      DEGRADE -> Output:MRNA;
      CAP->Ground;
      DC->Ground;
      VR->Ground;



}
